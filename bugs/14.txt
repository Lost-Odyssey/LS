## 实例代码

type Adder      = Block() -> int

fn sum_int_vec(vec(int) v) -> int {
    int s = 0
    int i = 0
    while i < v.length {
        s = s + v[i]
        i = i + 1
    }
    return s
}

fn main() {
    // E.1.1: closure captures vec(int), body passes to fn; verify correct value
    vec(int) nums = [10, 20, 30]
    Adder adder = || {
        return sum_int_vec(nums.copy())
    }
    int r1 = adder()
    print(r1)
    print(nums)
}
